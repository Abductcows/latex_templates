{\lt{
\begin{proposition}\label{basic}
Let $n,q$ and $A_j$ be positive integers with $A_j$ such that 
\begin{equation}\label{asterisq0}
A_j \in I_{j}=\bigg{(}\frac{q^{j/(n+1)+f_q(n)}}{2},\frac{q^{j/(n+1)+f_q(n)}}{1.5}\bigg{)},
\end{equation}
 with $1\leq j\leq n.$
The sequence $f_q(n):{\mathbb{N}}\rightarrow (0,1)$ is such that,
\begin{equation}\label{asterisq1}
f_q(n)+\frac{n}{n+1}<1 
\end{equation}
and
\begin{equation}\label{asterisq2}
\frac{q^{1+2f(n)}}{1.5}<q - \frac{1}{2}q^{n/(n+1)+f_q(n)}.
\end{equation}

With $L$ we denote the full rank lattice of rank $n+1$ generated by the vectors
${\bf b}_0 = (-1,A_1,...,A_{n}),$
${\bf b}_j = (0,0,...,q,...,0),$ where $q$ is in the position $j+1$ for $j=1,...,n.$ Then, for all non-zero ${\bf v}\in L$ we have
$$||{\bf v}||>\frac{q^{n/(n+1)+f_q(n)}}{2}.$$
\end{proposition}
}}
